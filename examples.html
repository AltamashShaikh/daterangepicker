daterangepicker attached to a button

Single mode

<input data-bind="
  daterangepicker: dateRange3,
  daterangepickerOptions: {
    single: true
  }
"/>

Month picker

<input data-bind="
  daterangepicker: dateRange4,
  daterangepickerFormat: 'MMMM YYYY',
  daterangepickerOptions: {
    single: true,
    periods: ['month']
  }
"/>

Standalone & Single

<div data-bind="
  daterangepicker: dateRange5,
  daterangepickerFormat: 'MMMM YYYY',
  daterangepickerOptions: {
    single: true,
    standalone: true
  }
"/>
